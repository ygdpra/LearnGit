const URL = 'http://zero.webappsecurity.com/index.html';
const SEARCHBOX = '#searchTerm';
const a = 'resultText';


class HomePage{
    static visit() {
        cy.visit(URL)
    }
    static fillSearchbox(zero) {
        cy.get(SEARCHBOX).type(zero + '{enter}')
    }
    static showResult(resultText) {
        cy.get(a).should('contain.text', zero)
    }

 }
 export default HomePage
